f_array_avg(array<float> arr) =>
    int n = array.size(arr)
    float result = na
    if n > 0
        float sum = 0.0
        for i = 0 to n - 1
            sum += array.get(arr, i)
        result := sum / n
    result

f_price(float x) =>
    na(x) ? "na" : str.tostring(x, format.mintick)

f_pct(float x) =>
    na(x) ? "na" : str.tostring(x * 100.0, "#.##") + "%"

f_num(float x) =>
    na(x) ? "na" : str.tostring(x, "#.###")

f_vol(float x) =>
    na(x) ? "na" :
     x >= 1000000 ? str.tostring(x / 1000000.0, "#.##") + "M" :
     x >= 1000 ? str.tostring(x / 1000.0, "#.#") + "K" :
     str.tostring(x, "#")

f_label_size(string s) =>
    s == "Tiny" ? size.tiny :
     s == "Small" ? size.small :
     s == "Large" ? size.large :
     size.normal

            lastReason := "$Vol15 below min."
        else if not upperSignal and not lowerSignal
            lastReason := "No upper/lower signal."
        else
            lastReason := "Core filter failed."

//=============================================================================
// Exit logic
// Exit checks start after entry bar.
// Same-minute TP/stop conflict resolves as STOP.
//=============================================================================

if active and bar_index > tradeEntryBar
    bool stopHit = tradeSide == 1 ? low <= tradeStop : high >= tradeStop
    bool targetHit = tradeSide == 1 ? high >= tradeTarget : low <= tradeTarget

    if stopHit
        stopSignal := true
        exitSide := tradeSide
        exitPrice := tradeStop

        lastStatus := "STOPPED"
        lastAction := "STOP"
        lastReason := "Stop touched. Same-minute conflict = stop."

        active := false

        if showLabels
            label.new(
                 x = bar_index,
                 y = exitSide == 1 ? low : high,
                 text = "STOP\n" + f_price(exitPrice),
                 style = exitSide == 1 ? label.style_label_down : label.style_label_up,
                 color = stopColor,
                 textcolor = labelTextColor,
                 size = f_label_size(labelSizeInput),
                 yloc = yloc.price)

        if enableDynamicAlerts
            alert(
                 syminfo.ticker + " STOP" +
                 " | " + tradeBranch +
                 " | Exit: " + f_price(exitPrice) +
                 " | Shares: " + str.tostring(tradeShares),
                 alert.freq_once_per_bar)

    else if targetHit
        takeProfitSignal := true
        exitSide := tradeSide
        exitPrice := tradeTarget

        lastStatus := "TAKE PROFIT"
        lastAction := "TP"
        lastReason := "Target touched."

        active := false

        if showLabels
            label.new(
                 x = bar_index,
                 y = exitSide == 1 ? high : low,
                 text = "TP\n" + f_price(exitPrice),
                 style = exitSide == 1 ? label.style_label_down : label.style_label_up,
                 color = tpColor,
                 textcolor = labelTextColor,
                 size = f_label_size(labelSizeInput),
                 yloc = yloc.price)

        if enableDynamicAlerts
            alert(
                 syminfo.ticker + " TAKE PROFIT" +
                 " | " + tradeBranch +
                 " | Exit: " + f_price(exitPrice) +
                 " | Shares: " + str.tostring(tradeShares),
                 alert.freq_once_per_bar)

    else if useTimeExit and inTimeExitWindow and barstate.isconfirmed
        timeExitSignal := true
        exitSide := tradeSide
        exitPrice := close

        lastStatus := "TIME EXIT"
        lastAction := "FLAT"
        lastReason := "No TP/stop before time exit."

        active := false

        if showLabels
            label.new(
                 x = bar_index,
                 y = close,
                 text = "TIME EXIT\n" + f_price(exitPrice),
                 style = label.style_label_left,
                 color = waitColor,
                 textcolor = textColor,
                 size = f_label_size(labelSizeInput),
                 yloc = yloc.price)

        if enableDynamicAlerts
            alert(
                 syminfo.ticker + " TIME EXIT" +
                 " | " + tradeBranch +
                 " | Exit: " + f_price(exitPrice) +
                 " | Shares: " + str.tostring(tradeShares),
                 alert.freq_once_per_bar)
